\section{Overview}
%==============================================================================
\textbf{Smart Fiatio} is a local web app: user uploads a portrait photo, picks a kurta from 31 designs, AI fits the kurta on the person (upper body). Stack: \textbf{React/Vite} UI $\rightarrow$ \textbf{FastAPI} $\rightarrow$ Hugging Face \textbf{IDM-VTON} (free). Garments credited to \href{https://www.affordable.pk/}{affordable.pk}. Replicate is \textbf{not} used.

%==============================================================================
\section{How It Works (Flow)}
%==============================================================================

\subsection{User flow}
\begin{enumerate}[label=\textbf{\arabic*.}]
  \item \textbf{Upload photo} (JPG/PNG, portrait, front view) $\rightarrow$ validated in browser.
  \item \textbf{Select kurta} from gallery $\rightarrow$ large preview updates.
  \item \textbf{Run Try-On} $\rightarrow$ progress 0--100\%.
  \item \textbf{Output} $\rightarrow$ view result, \textbf{Download picture}.
\end{enumerate}

\subsection{Technical flow (one try-on)}
\begin{verbatim}
[User] upload photo + pick kurta
    |
    v
[Browser] validate portrait -> JPEG resize (imageUtils.ts)
    |
    v
[Browser] POST /api/tryon  (FormData: human_img, garm_img, garment_des)
    |        via Vite proxy :5173 -> :8765
    v
[FastAPI] replicate_tryon_server.py -> hf_tryon.py
    |
    v
[HF Space] yisol/IDM-VTON  (gradio_client, ~1-5 min queue)
    |        upper-body mask + kurta image + text prompt
    v
[API] returns { url } or { image_base64 }
    |
    v
[Browser] show image + download
\end{verbatim}

\section{Run the Project}
%==============================================================================
\textbf{Prerequisites:} Node 18+, Python 3.10+, internet.

\begin{lstlisting}
pip install -r requirements.txt && npm install
\end{lstlisting}

\textbf{Terminal A:} \texttt{npm run api} $\rightarrow$ \url{http://127.0.0.1:8765/health}

\textbf{Terminal B:} \texttt{npm run dev} $\rightarrow$ \url{http://localhost:5173/}

\textbf{Add kurtas:} \texttt{npm run sync:garments} then refresh browser.

\textbf{.env:} \texttt{TRYON\_PROVIDER=idm-vton}, optional \texttt{HF\_TOKEN=} for better HF queue.

\subsection{API (POST \texttt{/api/tryon})}
\texttt{human\_img}, \texttt{garm\_img}, \texttt{garment\_des} (optional), \texttt{tryon\_mode=upper}. Response: \texttt{url} or \texttt{image\_base64}.

\begin{tabular}{@{}p{3.2cm}p{10cm}@{}}
\toprule
Issue & Fix \\
\midrule
API error & Run \texttt{npm run api} \\
No kurtas & \texttt{npm run sync:garments} \\
HF timeout & Retry; add \texttt{HF\_TOKEN} in \texttt{.env} \\
Wrong port & Use Vite URL from terminal (5173 or 5174) \\
